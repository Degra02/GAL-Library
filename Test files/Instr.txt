Matrice RiduzioneMatrice(Matrice *m, int riga, int colonna){
    Matrice md(m->nr - 1, m->nc - 1);

    int counterR = 0, counterC = 0;

    for (int i = 0; i < m->nr; i++){
        for (int j = 0; j < m->nc; j++){
            if (i != riga && j != colonna){
                if (counterC == md.nc){
                    counterC = 0;
                    counterR++;
                }
                md.mat[counterR][counterC++] = m->mat[i][j];
            }
        }
    }

    return md;
}

float det(Matrice *m){
